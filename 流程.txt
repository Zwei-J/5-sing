CREATE TABLE user_comment(
	ucid INT PRIMARY KEY AUTO_INCREMENT,
	sid INT,
		FOREIGN KEY(sid) REFERENCES songlist(sid),
	comment VARCHAR(512) NOT NULL,
	cuser VARCHAR(32),
		FOREIGN KEY(cuser) REFERENCES wx_user(uname),
	time DATETIME
);
